// settings.js — flyt dette ind hvis du foretrækker ekstern fil const nameInput = document.getElementById("name"); const intervalSelect = document.getElementById("interval"); const saveBtn = document.getElementById("saveBtn"); nameInput.value = localStorage.getItem("userName") || ""; intervalSelect.value = localStorage.getItem("pp_interval_min") || "40"; saveBtn.onclick = () => { localStorage.setItem("userName", nameInput.value.trim()); localStorage.setItem("pp_interval_min", intervalSelect.value); alert("Indstillinger gemt"); };
